\subsection{The NKP Action}
The theory is defined by the action
\begin{equation}
S[\rho, g] = \int d^4x \, \sqrt{-g} \left[ \frac{M^2}{2} \frac{(\nabla \rho)^2}{\rho^2} - V(\rho) - J(\mathbf{x})\rho \right],
\end{equation}
where the kinetic term is equivalent to \(\frac{M^2}{2} (\nabla \ln \rho)^2\).

The potential is
\begin{equation}
V(\rho) = \frac{\alpha_0}{2} (\rho - 1)^2 + \lambda \tanh(\beta (\rho - 1)),
\end{equation}
with effective stiffness
\begin{equation}
\alpha_{\rm eff}(\rho) = \alpha_0 \bigl[1 + \lambda \tanh(\beta |\ln \rho|)\bigr].
\end{equation}

The source term couples matter defects linearly:
\begin{equation}
L_{\rm src} = -J(\mathbf{x})\rho.
\end{equation}

\subsection{Weak-Field GR Limit}
In the weak-field expansion \(\rho = 1 + \delta\rho\) (\(|\delta\rho| \ll 1\)), the metric becomes
\[
g_{00}^{\rm eff} \approx -(1 - 2\delta\rho), \quad g_{ij}^{\rm eff} \approx (1 - 2\delta\rho)\delta_{ij}.
\]
Identifying the Newtonian potential
\begin{equation}

\end{equation}

\[
